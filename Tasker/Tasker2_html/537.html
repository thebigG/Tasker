38     /**
39      * @brief AudioDevice::AudioDevice
40      * @param newFormat
41      */
42     AudioDevice::AudioDevice(const QAudioFormat newFormat)
43     : qAudioFormat(newFormat), deviceLevel(0.0), minAmplitude(0.0), maxAmplitude(0) {
44         switch (qAudioFormat.sampleSize()) {
45         case 8:
46             switch (qAudioFormat.sampleType()) {
47             case QAudioFormat::UnSignedInt:
48                 maxAmplitude = 255;
49                 break;
50             case QAudioFormat::SignedInt:
51                 maxAmplitude = 127;
52                 break;
53             default:
54                 break;
55             }
56     
57             break;
58     
59         case 16:
60             switch (qAudioFormat.sampleType()) {
61             case QAudioFormat::UnSignedInt:
62                 maxAmplitude = 65535;
63                 break;
64             case QAudioFormat::SignedInt:
65                 maxAmplitude = 32767;
66                 break;
67             default:
68                 break;
69             }
70     
71             break;
72     
73         case 32:
74             switch (qAudioFormat.sampleType()) {
75             case QAudioFormat::UnSignedInt:
76                 maxAmplitude = 0xffffffff;
77                 break;
78             case QAudioFormat::SignedInt:
79                 maxAmplitude = 0x7fffffff;
80                 break;
81             case QAudioFormat::Float:
82                 maxAmplitude = 0x7fffffff; // Kind of
83                 break;
84             default:
85                 break;
86             }
87     
88             break;
89     
90         default:
91             break;
92         }
93     }

155     /**
156      * @brief AudioDevice::writeData
157      * @param data
158      * @param len
159      * @return
160      */
161     qint64 AudioDevice::writeData(const char *data, qint64 len) {
162         qreal captureValue = 0.0;
163     
164         if (maxAmplitude) {
165             Q_ASSERT(qAudioFormat.sampleSize() % 8 == 0);
166     
167             const qint64 channelBytes = (qAudioFormat.sampleSize() / 8);
168             const qint64 sampleBytes = (qAudioFormat.channelCount() * channelBytes);
169     
170             Q_ASSERT(len % sampleBytes == 0);
171     
172             const qint64 numSamples = (len / sampleBytes);
173     
174             quint32 maxValue = 0;
175             const unsigned char *ptr = reinterpret_cast<const unsigned char *>(data);
176     
177             for (int i = 0; i < numSamples; ++i) {
178                 for (int j = 0; j < qAudioFormat.channelCount(); ++j) {
179                     quint32 value = 0;
180     
181                     if (qAudioFormat.sampleSize() == 8 &&
182                         qAudioFormat.sampleType() == QAudioFormat::UnSignedInt) {
183                         value = *reinterpret_cast<const quint8 *>(ptr);
184                     } else if (qAudioFormat.sampleSize() == 8 &&
185                                qAudioFormat.sampleType() == QAudioFormat::SignedInt) {
186                         value = static_cast<quint32>(
187                             qAbs(*reinterpret_cast<const qint8 *>(ptr)));
188                     } else if (qAudioFormat.sampleSize() == 16 &&
189                                qAudioFormat.sampleType() == QAudioFormat::UnSignedInt) {
190                         if (qAudioFormat.byteOrder() == QAudioFormat::LittleEndian) {
191                             value = qFromLittleEndian<quint16>(ptr);
192                         } else {
193                             value = qFromBigEndian<quint16>(ptr);
194                         }
195                     } else if (qAudioFormat.sampleSize() == 16 &&
196                                qAudioFormat.sampleType() == QAudioFormat::SignedInt) {
197                         if (qAudioFormat.byteOrder() == QAudioFormat::LittleEndian) {
198                             value = static_cast<quint32>(qAbs(qFromLittleEndian<qint16>(ptr)));
199                         } else {
200                             value = static_cast<quint32>(qAbs(qFromBigEndian<qint16>(ptr)));
201                         }
202                     } else if (qAudioFormat.sampleSize() == 32 &&
203                                qAudioFormat.sampleType() == QAudioFormat::UnSignedInt) {
204                         if (qAudioFormat.byteOrder() == QAudioFormat::LittleEndian) {
205                             value = qFromLittleEndian<quint32>(ptr);
206                         } else {
207                             value = qFromBigEndian<quint32>(ptr);
208                         }
209                     } else if (qAudioFormat.sampleSize() == 32 &&
210                                qAudioFormat.sampleType() == QAudioFormat::SignedInt) {
211                         if (qAudioFormat.byteOrder() == QAudioFormat::LittleEndian) {
212                             value = static_cast<quint32>(qAbs(qFromLittleEndian<qint32>(ptr)));
213                         } else {
214                             value = static_cast<quint32>(qAbs(qFromBigEndian<qint32>(ptr)));
215                         }
216                     } else if (qAudioFormat.sampleSize() == 32 &&
217                                qAudioFormat.sampleType() == QAudioFormat::Float) {
218                         value = static_cast<quint32>(
219                             qAbs(*reinterpret_cast<const float *>(ptr) * 0x7fffffff)); // assumes 0-1.0
220                     }
221     
222                     maxValue = qMax(value, maxValue);
223                     ptr += channelBytes;
224                 }
225             }
226     
227             maxValue = getMin(maxValue, maxAmplitude);
228             captureValue = (qreal(maxValue) / maxAmplitude);
229             deviceLevel = captureValue - minAmplitude;
230         }
231     //    qDebug() << "AudioDevice writeData() thread id: << " << QThread::currentThreadId();
232         qDebug()<<"device level:" << deviceLevel;
233         QThread::sleep(2);
234         return len;
235     }
